Bibliotexas são conjuntos de códigos que podemos reutilizar.
Exustem muitas bibliotecas(Infinada de bibliotecas).

import random

informeMinimo = int(input("Informe o valor inicial: "))
informeMaximo = int(input("Informe o valor máximo: "))

numeroAleatorio = random.randint(informeMinimo, informeMaximo)
print("Número gerado: ", numeroAleatorio)

sucessor = numeroAleatorio + 1
antescessor = numeroAleatorio - 1
print("O sucessor de ", numeroAleatorio, " é: ", sucessor)
print("O antescessor de ", numeroAleatorio, " é: ", antescessor)
